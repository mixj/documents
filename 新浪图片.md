## 抓取新浪的各种图片

* 要用浏览器调试中的手机模式

### 评论
```
var basr_url = 'https://m.weibo.cn/comments/hotflow?'
var id = $render_data.status.id;
var params = `id=${id}&mid=${id}&max_id_type=0`
var arr = [];
var md = ''
var index = 0
var reg = new RegExp('.cn/large/([^/]+).jpg" data-hid','ig');

function loadXMLDoc(url)
{
    var xmlhttp;
        //  IE7+, Firefox, Chrome, Opera, Safari 浏览器执行代码
    xmlhttp=new XMLHttpRequest();
    xmlhttp.onreadystatechange=function()
    {
        if (xmlhttp.readyState==4 && xmlhttp.status==200)
        {
            let {data:{data,max_id,max_id_type}} = JSON.parse(xmlhttp.responseText)
            data.forEach(item=>{
                let {pic:{pid,url}={pid:null},comments} = item
                if(pid && !arr.includes(pid)){
                        index++;
                        arr.push(pid)
                        md+=`![](${url})<br/>`
                }
				if(comments && comments.length){
                    comments.forEach(item=>{
                        let {text} = item;
                        while((result = reg.exec(text)) != null) {
                            arr.push(result[1])
                        }
                    })
                }
            })
            var searchParams = new URLSearchParams(params)
            searchParams.set('max_id', max_id)
            searchParams.set('max_id_type', max_id_type)

            var getUrl = `${basr_url}${searchParams.toString()}`
            if(max_id){
                setTimeout(()=>{
                    loadXMLDoc(getUrl)
                },2000)
            }else{
                console.log(JSON.stringify(arr))
            }
        }
    }
    xmlhttp.open("GET",url,true);
    xmlhttp.send();
}

start = `${basr_url}${params}`
loadXMLDoc(start)
```

```
var basr_url = 'https://m.weibo.cn/api/container/getIndex?'
var params = window.location.href.split('?')[1]
var reg = new RegExp('.cn/large/([^/]+).jpg" data-hid','ig');
var arr = []
function loadXMLDoc(url)
{
    var xmlhttp;
        //  IE7+, Firefox, Chrome, Opera, Safari 浏览器执行代码
    xmlhttp=new XMLHttpRequest();
    xmlhttp.onreadystatechange=function()
    {
        if (xmlhttp.readyState==4 && xmlhttp.status==200)
        {
        
            let {data:{cards,pageInfo:{since_id}}} = JSON.parse(xmlhttp.responseText)
            let searchParams = new URLSearchParams(params)
            searchParams.set('since_id', since_id)
            let getUrl = `${basr_url}${searchParams.toString()}`
            cards.forEach(item=>{
                let {itemid, card_group} = item
                if(card_group && card_group.length){
                    card_group.forEach(ele=>{
                        let { mblog:{pics} } = ele
                        if(pics && pics.length){
                            pics.forEach(pic=>{
                                let {pid} = pic
                                arr.push(pid)
                            })
                        }
                    })
                }
            })
            if(since_id){
                setTimeout(()=>{
                    loadXMLDoc(getUrl)
                },2000)
            }else{
                console.log(JSON.stringify(arr))
            }
        }
    }
    xmlhttp.open("GET",url,true);
    xmlhttp.send();
}

start = `${basr_url}${params}`
loadXMLDoc(start)
```
